# Redirect all HTTP requests to HTTPS.
<VirtualHost *:80>
    ServerAdmin admin@#DOMAIN#
    ServerName #DOMAIN#
    Redirect permanent / https://#DOMAIN#/
</VirtualHost>

<VirtualHost *:443>
    ServerAdmin admin@#DOMAIN#
    ServerName #DOMAIN#

    DocumentRoot "#DOCROOT#"
    <Directory "#DOCROOT#">
        Options Indexes FollowSymLinks MultiViews
        AllowOverride All
        Require all granted
    </Directory>

    ErrorLog "#LOGDIR#/#SHORTNAME#/error_log"
    LogLevel warn
    CustomLog "#LOGDIR#/#SHORTNAME#/access_log" combined

    SSLEngine On
    SSLProtocol all -SSLv2 -SSLv3
    SSLCipherSuite ALL:!ADH:!EXPORT:!SSLv2:RC4+RSA:+HIGH:+MEDIUM:+LOW

    SSLCertificateFile "/etc/apache2/certs/_wildcard.local.stad.gent.pem"
    SSLCertificateKeyFile "/etc/apache2/certs/_wildcard.local.stad.gent-key.pem"
</VirtualHost>
